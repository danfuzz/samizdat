# Intlets.
i1 := @0;
i2 := @-1;
i3 := @ffff;
i4 := @-ffff;
i5 := @123456789abcdef00fedcba987654321aabbccddeeff;
i6 := @12345;

# Listlets.
l1 := @[];
l2 := @[@[]];
l3 := @[@0, @1, @2, @3];

# String-form intlets.
s1 := @""; # == @[]
s1 := @"foo";
s2 := @"bar";
s3 := @[@62, @61, @7a]; # == @"baz"

# Maplets.
m1 := @{};
m2 := @{@0: @a, @1: @b, @2: @c, @[@1]: @ddd, @"blort": @"zorch"};

# Uniquelets.
u1 := @@;
u2 := @@;
u3 := @@;

# Value references.
v1 := i1;
v2 := @[i1, i2, i3];
v3 := @{v1: v2, v2: v1};
